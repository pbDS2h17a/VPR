########Ordner erstellen/initialisieren########
cd VPR_RISIKO
git init
git remote add source https://github.com/pbDS2h17a/VPR.git //Source linken
git remote -v //source links einsehen
git pull --all //Alle Daten laden
########Branch erstellen#########
git checkout Development //in  Development wechseln
git branch pbds2h17axy //Branch in Development erstellen
git checkout pbds2h17axy //in Branch wechseln
########Dateien erstellen/kopieren etc.########
Beispiel:
vim Beispiel.txt //Erstellt Datei (Daten eintragen)
git add . ####ODER#### git add <Datei> //fügt neue Dateien der Branch hinzu
git commit -m "Beispiel.txt hinzugefügt" //Datei übernehmen und optionale Message beifügen
git checkout ####SHIFT+TAB für Übersicht####
git merge pbds2h17axy //Branch im Development hinzufügen/updaten
git push pbds2h17axy //Branch-Übernahme und hinzufügen bestätigen
